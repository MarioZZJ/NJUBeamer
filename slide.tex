\section{模板介绍}
% ==============================

\begin{frame}{关于本模板}
  \begin{itemize}
    \item 这是南京大学 Beamer 演示文稿模板
    \item 配色基于南京大学视觉形象规范化标准
    \item 使用 Xe\LaTeX{} 编译，支持中文
    \item 项目地址：\url{https://github.com/nju-lug/NJUBeamer}
  \end{itemize}
\end{frame}

\begin{frame}{主题选项}
  \begin{block}{基本用法}
    在文档导言区使用：
    \texttt{\textbackslash usetheme\{njubeamer\}}
  \end{block}
  \begin{block}{可选参数}
    \begin{description}
      \item[\texttt{en}] 纯英文模式（不加载 ctex 宏包）
      \item[\texttt{sectiontoc}] 每节前自动显示目录
      \item[\texttt{nosectionpage}] 关闭节过渡页
    \end{description}
  \end{block}
\end{frame}

    \[
      J(\theta) = \mathbb{E}_{\pi_\theta}[G_t]

  \end{block}
